function BaseClass(super)
	local super = super
	if super ~= nil then
		super.__index = super
	else
		super = {}
		super.Ctor = function(obj, ...) end
		super.Dtor = function(obj, ...) end
		super.__index = super
	end

	local obj = {
		Ctor = function(obj, ...) end,
		Dtor = function(obj, ...) end,
		super = super
	}

	obj.New = function(...)
		BaseCtor(obj, obj, ...)
		
		return BaseClass(obj)
	end
	setmetatable(obj, super)

	return obj
end

function BaseCtor(obj, super, ...)
	if super ~= nil then
		BaseCtor(obj, super.super, ...)
		super:Ctor(obj, ...)
	end
end
